public Task Execute(IJobExecutionContext context)
	{
		var diretorioTemporario = _configuracaoDiretorioTemporario.Diretorio;
		var apagarAquivoDias = _configuracaoDiretorioTemporario.ApagarAquivoDias;
		try
		{
			if (Directory.Exists(diretorioTemporario))
			{
				foreach (var dir in Directory.GetDirectories(diretorioTemporario))
				{
					if (Directory.GetFiles(dir).Length.Equals(0))
						Directory.Delete(dir);
				}

				DeletarArquivos(diretorioTemporario, apagarAquivoDias);
			}
		}
		catch (Exception e)
		{
			_logger.LogError(e, $"Falha aoa deletar arquivos tempor?rios");
		}
		
		return Task.CompletedTask;
	}


	public void DeletarArquivos( string diretorioTemporario, int apagarAquivoDias)
	{

		try
		{
			foreach (var file in Directory.GetFiles(diretorioTemporario, "*.*", SearchOption.AllDirectories))
			{
				DateTime DateTimeFile = File.GetCreationTime(file);
				int tempoFileDay = DateTime.Now.Subtract(DateTimeFile).Days;

				if (tempoFileDay > apagarAquivoDias)
				{
					File.Delete(file);
				}

			}
		}
		catch (Exception e)
		{
			_logger.LogError(e, $"Falha aoa deletar arquivos tempor?rios");
		}
	}

}
